  "type": "mcq | short_answer | numeric | long_answer", // Question response type enum
  "question_text": "String", // Prompt shown to the learner
  "metadata": { // Type-specific grading/configuration object
    "options": [ // Required when type = "mcq"
      { "id": "A", "text": "Option A" }, // Option id (submitted value) and visible label
      { "id": "B", "text": "Option B" }
    ],
    "correct_answer": "A", // Required when type = "mcq"; must match an option id
    "accepted_answers": ["answer1", "answer2"], // Required when type = "short_answer"

    "ignore_whitespace": true, // For short answers: true strips/ignores spaces before compare
    "numeric_answer": 42, // Required when type = "numeric"
    "tolerance": 0.01 // Optional numeric margin of error for numeric grading
  },
  "difficulty": "1 | 2 | 3", // 1=easy, 2=intermediate, 3=advanced
  "topic": "String", // Subject area for grouping/reporting
  "skillCategory": "1 | 2 | 3", // 1=recall, 2=conceptual, 3=memorization
  "explanation": "String", // Optional rationale shown after answering/review
  "createdAt": "FieldValue.serverTimestamp()", // Server-side creation timestamp
  "lastModified": "FieldValue.serverTimestamp()", // Server-side update timestamp
  "isArchived": false // Soft-delete flag; archived questions are excluded from active use
}

// general structure
// [
//   {
//     "type": "mcq | short_answer | numeric | long_answer",
//     "question_text": "string",
//     "metadata": { ...type-specific fields... },
//     "difficulty": 1 | 2 | 3,
//     "topic": "string",
//     "skillCategory": 1 | 2 | 3,
//     "explanation": "string (optional)"
//   }
// ]

// for mcq
// "type": "mcq",
// "metadata": {
//   "options": [{ "id": "A", "text": "..." }],
//   "correct_answer": "A"
// }

// for short answer
// "type": "short_answer",
// "metadata": {
//   "accepted_answers": ["stack", "a stack"],
//   "case_sensitive": false,
//   "ignore_whitespace": true
// }

// for numeric
// "type": "numeric",
// "metadata": {
//   "numeric_answer": 3.14,
//   "tolerance": 0.01
// }

// for long answer
// "type": "long_answer",
// "metadata": {}

// example
{"questions": [
  {
    "type": "mcq",
    "question_text": "What is the time complexity of binary search?",
    "metadata": {
      "options": [
        { "id": "A", "text": "O(n)" },
        { "id": "B", "text": "O(log n)" },
        { "id": "C", "text": "O(n log n)" },
        { "id": "D", "text": "O(1)" }
      ],
      "correct_answer": "B"
    },
    "difficulty": 1,
    "topic": "Algorithms",
    "skillCategory": 2,
    "explanation": "Binary search halves the search space each iteration."
  }
]}
